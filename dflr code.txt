# ### 2. Helper functions for DFLR calculation

def compute_record_fidelity(before_df, after_df, key_column):
    """
    Compute record-level fidelity.
    before_df, after_df: DataFrames with a common unique key column.
    Returns: record_fidelity (%), missing_in_after, extra_in_after.
    """
    before_keys = set(before_df[key_column].astype(str))
    after_keys = set(after_df[key_column].astype(str))
    
    common = before_keys.intersection(after_keys)
    record_fidelity = len(common) / len(before_keys) * 100 if len(before_keys) > 0 else 0
    
    missing = before_keys - after_keys
    extra = after_keys - before_keys
    return record_fidelity, missing, extra

def compute_parameter_fidelity(before_df, after_df, key_column, parameters):
    """
    For each parameter, compute the percentage of records where the value matches exactly.
    Only considers records that exist in both DataFrames.
    Returns dict: {parameter: fidelity_percentage}
    """
    # Merge on key column
    merged = pd.merge(before_df, after_df, on=key_column, suffixes=('_before', '_after'))
    fidelity = {}
    for param in parameters:
        if param+'_before' in merged.columns and param+'_after' in merged.columns:
            # Compare values as strings to avoid type issues
            before_vals = merged[param+'_before'].astype(str)
            after_vals = merged[param+'_after'].astype(str)
            matches = (before_vals == after_vals).sum()
            total = len(merged)
            fidelity[param] = (matches / total * 100) if total > 0 else 0
        else:
            fidelity[param] = 0.0
    return fidelity

def compute_overall_dflr(before_df, after_df, key_column, parameters):
    """
    Compute overall weighted DFLR:
    - total cells = (number of records in before) * (number of parameters)
    - matched cells = sum over parameters of (number of matching records for that parameter)
    DFLR = 1 - (matched_cells / total_cells)
    """
    # Merge on key to get common records only (conservative approach)
    merged = pd.merge(before_df, after_df, on=key_column, suffixes=('_before', '_after'))
    total_records = len(merged)
    total_cells = total_records * len(parameters)
    matched_cells = 0
    for param in parameters:
        if param+'_before' in merged.columns and param+'_after' in merged.columns:
            before_vals = merged[param+'_before'].astype(str)
            after_vals = merged[param+'_after'].astype(str)
            matched_cells += (before_vals == after_vals).sum()
    dflr = (1 - matched_cells / total_cells) * 100 if total_cells > 0 else 100
    return dflr

def plot_fidelity(fidelity_dict, title):
    """Plot parameter fidelity as a horizontal bar chart."""
    params = list(fidelity_dict.keys())
    values = list(fidelity_dict.values())
    plt.figure(figsize=(10, 6))
    bars = plt.barh(params, values, color='skyblue')
    plt.xlabel('Fidelity (%)')
    plt.title(title)
    plt.xlim(0, 100)
    for bar, val in zip(bars, values):
        plt.text(val + 1, bar.get_y() + bar.get_height()/2, f'{val:.1f}%', va='center')
    plt.tight_layout()
    plt.show()

# Upload before-export door schedule (e.g., door_schedule_before.csv)
print("Upload BEFORE export door schedule (CSV or Excel):")
door_before_file = files.upload()
door_before = None
for fn in door_before_file.keys():
    if fn.endswith('.csv'):
        door_before = pd.read_csv(io.BytesIO(door_before_file[fn]))
    else:
        door_before = pd.read_excel(io.BytesIO(door_before_file[fn]), sheet_name=0)
    break

# Upload after-export door data (CSV extracted from IFC)
print("Upload AFTER export door data (CSV):")
door_after_file = files.upload()
door_after = None
for fn in door_after_file.keys():
    door_after = pd.read_csv(io.BytesIO(door_after_file[fn]))
    break

# Define key column and parameters for doors
DOOR_KEY = 'Mark'           # unique identifier
DOOR_PARAMS = ['Level', 'Rough Height', 'Rough Width', 'Family', 'Type']

# Show first few rows
print("Before export (first 5 rows):")
print(door_before.head())
print("\nAfter export (first 5 rows):")
print(door_after.head())

# Compute DFLR
record_fid, missing, extra = compute_record_fidelity(door_before, door_after, DOOR_KEY)
param_fid = compute_parameter_fidelity(door_before, door_after, DOOR_KEY, DOOR_PARAMS)
overall_dflr = compute_overall_dflr(door_before, door_after, DOOR_KEY, DOOR_PARAMS)

print("\n=== DOORS DFLR RESULTS ===")
print(f"Record fidelity: {record_fid:.2f}%")
print(f"Missing records (in after): {missing}")
print(f"Extra records (in after): {extra}")
print(f"Overall DFLR: {overall_dflr:.2f}%")
print("\nParameter-level fidelity:")
for p, f in param_fid.items():
    print(f"  {p}: {f:.2f}%")

plot_fidelity(param_fid, "Door Parameter Fidelity")
